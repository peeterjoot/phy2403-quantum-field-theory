\makeproblem{}{problem:psq:1}{
Show that
\begin{equation*}
\lr{
   p^\beta {p'}^\alpha
   +
   p^\alpha {p'}^\beta
   -
   p \cdot p'
   g^{\alpha\beta}
}
\times
\lr{
   {k'}_\beta
   {k}_\alpha
   +
   {k'}_\alpha
   {k}_\beta
   -
   \lr{
   k \cdot k'
   +
      m_\mu^2
   }
   g_{\alpha\beta}
}
=
2
\lr{
   p \cdot k p' \cdot k'
   +
   p \cdot k' p' \cdot k
   +
   m_\mu^2 p \cdot p'
}
\end{equation*}
} % problem

\makeanswer{problem:psq:1}{
Proceeding mechanically, but carefully, we have
\begin{equation}\label{eqn:psq:40}
\begin{aligned}
&  p^\beta {p'}^\alpha
   {k'}_\beta
   {k}_\alpha
+
   p^\beta {p'}^\alpha
   {k'}_\alpha
   {k}_\beta
-
   p^\beta {p'}^\alpha
   \lr{
      k \cdot k'
      +
      m_\mu^2
   }
   g_{\alpha\beta} \\
&\qquad +
   p^\alpha {p'}^\beta
   {k'}_\beta
   {k}_\alpha
+
   p^\alpha {p'}^\beta
   {k'}_\alpha
   {k}_\beta
-
   p^\alpha {p'}^\beta
   \lr{
      k \cdot k'
      +
      m_\mu^2
   }
   g_{\alpha\beta} \\
&\qquad -
   p \cdot p'
   g^{\alpha\beta}
   {k'}_\beta
   {k}_\alpha
-
   p \cdot p'
   g^{\alpha\beta}
   {k'}_\alpha
   {k}_\beta
+
   p \cdot p'
   g^{\alpha\beta}
   \lr{
      k \cdot k'
      +
      m_\mu^2
   }
   g_{\alpha\beta} \\
&=
p \cdot k' p' \cdot k
+
p \cdot k p' \cdot k'
-
p \cdot p'
   \lr{
      k \cdot k'
      +
      m_\mu^2
   } \\
&\qquad +
p \cdot k p' \cdot k'
+
p \cdot k' p' \cdot k
-
p \cdot p'
   \lr{
      k \cdot k'
      +
      m_\mu^2
   } \\
&\qquad -
   p \cdot p'
k \cdot k'
-
   p \cdot p'
k \cdot k'
+
4 p \cdot p'
   \lr{
      k \cdot k'
      +
      m_\mu^2
   } \\
&=
  2 p \cdot k' p' \cdot k
+ 2 p \cdot k p' \cdot k'
- 2 p \cdot p' k \cdot k'
+ 2 p \cdot p' \lr{ k \cdot k' + m_\mu^2 } \\
&=
  2 p \cdot k' p' \cdot k
+ 2 p \cdot k p' \cdot k'
+ 2 p \cdot p' m_\mu^2  \\
&=
2 \lr{
    p \cdot k' p' \cdot k
+   p \cdot k p' \cdot k'
+   p \cdot p' m_\mu^2
}.
\end{aligned}
\end{equation}
} % answer
