\makeproblem{Dirac Hamiltonian w/o zero-time field substitution.}{problem:qftLecture21b:1}{
} % problem

\makeanswer{problem:qftLecture21b:1}{
With time left in the mix the fields are
\begin{equation}\label{eqn:qftLecture21b:1180}
\begin{aligned}
\Psi(x)
&=
\sum_{s = 1}^2
\int \frac{d^3 p}{(2 \pi)^3 \sqrt{ 2 \omega_\Bp } }
\lr{
   e^{-i p \cdot x} u^s_\Bp a_\Bp^s
+
   e^{i p \cdot x} v^s_\Bp b_\Bp^s
} \\
\Psi^\dagger(x)
&=
\sum_{r = 1}^2
\int \frac{d^3 q}{(2 \pi)^3 \sqrt{ 2 \omega_\Bq } }
\lr{
   e^{i q \cdot x} u^{r\dagger}_\Bq a_\Bq^{r \dagger}
+
   e^{-i q \cdot x} v^{r\dagger}_\Bq b_\Bq^{r \dagger}
},
\end{aligned}
\end{equation}
and the Hamiltonian is
\begin{equation}\label{eqn:qftLecture21b:1200}
\begin{aligned}
H_{\text{Dirac}}
&=
\sum_{r,s = 1}^2
\int \frac{d^3 x d^3 p d^3 q}{(2 \pi)^6 2 \sqrt{ \omega_\Bp \omega_\Bq } }
\lr{
   e^{i q \cdot x} u^{r \dagger}_\Bq a_\Bq^{r \dagger}
+
   e^{-i q \cdot x} v^{r \dagger}_\Bq b_\Bq^{r \dagger}
}
\omega_\Bp
\lr{
   e^{-i p \cdot x} u^s_\Bp a_\Bp^s
-
   e^{i p \cdot x} v^s_\Bp b_\Bp^s
} \\
&=
\sum_{r,s = 1}^2
\int \frac{d^3 x d^3 p d^3 q}{(2 \pi)^6 2 \sqrt{ \omega_\Bp \omega_\Bq } }
\lr{
   e^{i \omega_\Bq t - i \Bq \cdot \Bx }
   u^{r \dagger}_\Bq a_\Bq^{r \dagger}
+
   e^{-i \omega_\Bq t + i \Bq \cdot \Bx }
   v^{r \dagger}_\Bq b_\Bq^{r \dagger}
}
\omega_\Bp
\lr{
   e^{-i \omega_\Bp t + i \Bp \cdot \Bx }
   u^s_\Bp a_\Bp^s
-
   e^{i \omega_\Bp t - i \Bp \cdot \Bx }
   v^s_\Bp b_\Bp^s
} \\
&=
\sum_{r,s = 1}^2
\int \frac{d^3 x d^3 p d^3 q}{(2 \pi)^6 2 \sqrt{ \omega_\Bp \omega_\Bq } }
\lr{
   e^{i \omega_\Bq t - i \Bq \cdot \Bx }
   u^{r \dagger}_\Bq a_\Bq^{r \dagger}
+
   e^{-i \omega_\Bq t - i \Bq \cdot \Bx }
   v^{r \dagger}_{-\Bq} b_{-\Bq}^{r \dagger}
}
\omega_\Bp
\lr{
   e^{-i \omega_\Bp t + i \Bp \cdot \Bx }
   u^s_\Bp a_\Bp^s
-
   e^{i \omega_\Bp t + i \Bp \cdot \Bx }
   v^s_{-\Bp} b_{-\Bp}^s
} \\
&=
\sum_{r,s = 1}^2
\int \frac{d^3 p }{(2 \pi)^3 2 }
\lr{
   e^{i \omega_\Bq t }
   u^{r \dagger}_\Bq a_\Bq^{r \dagger}
+
   e^{-i \omega_\Bq t }
   v^{r \dagger}_{-\Bq} b_{-\Bq}^{r \dagger}
}
\lr{
   e^{-i \omega_\Bp t }
   u^s_\Bp a_\Bp^s
-
   e^{i \omega_\Bp t }
   v^s_{-\Bp} b_{-\Bp}^s
} \\
&=
\sum_{r,s = 1}^2
\int \frac{d^3 p }{(2 \pi)^3 2 }
\lr{
   u^{r \dagger}_\Bq
   u^s_\Bp
   a_\Bq^{r \dagger}
   a_\Bp^s
-
   v^{r \dagger}_{-\Bq}
   v^s_{-\Bp}
   b_{-\Bq}^{r \dagger}
   b_{-\Bp}^s
},
\end{aligned}
\end{equation}
where a \( \deltathree(\Bp-\Bq) \) was factored out and evaluated, and the remaining \( v^{r\dagger}_{-\Bp} u^s, u^{r \dagger} v^s_{-\Bp} \) terms were killed off.
A final use of \cref{eqn:qftLecture21b:1100} completes the proof.
} % answer
